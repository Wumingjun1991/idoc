事件的绑定 # 

<!--绑定事件指令,这种方法在webstorm中会报红-->
<!--<div id="app" v-on:click="parent">-->
<!--可以简写成下面这种方式-->
<!--$event: 事件对象,通过这种方式传进去-->
<!--或者在绑定的函数中设置一个形参来接收事件对象
 parent(e){
                console.log(e);
                console.log("parent");
            }
-->
<div id="app" @click.self="parent($event)">
    <div @click="fn1">1</div>
    <div @click="fn2">2</div>
    <div @click="fn3">3</div>
    <a href="www.baidu.com" @click.prevent="aa">baidu</a>
</div>
<script src="js/vue2.0.js"></script>
<script>
    var v1 = new Vue({
        el:'#app',
        /*data:{
            fn1:function () {
                console.log("div1");
            },
            fn2:function () {
                console.log("div2");
            },
            fn3:function () {
                console.log("div3");
            },
            parent:function () {
                console.log("parent");
            }
        },*/
        /*methods:vue中专门用来存放函数的一个对象,函数写在这里面不用加关键字"functhin"*/
        methods:{
            fn1(){
                console.log("div1");
            },
            fn2(){
                console.log("div2");
            },
            fn3(){
                console.log("div3");
            },
            parent(e){
                console.log(e);
                console.log("parent");
            },
            aa(){
                alert(1)
            }
        }
    });
    // 在这里也会触发事件冒泡
    // 阻止事件冒泡的方法:
    // 1.给上一级事件加一个修饰符self: @click.self(只有点自己的时候才会触发绑定函数)
    // 2.给当前事件增加一个修饰符stop: @click.stop(阻止事件冒泡)
    // @click.prevent: 阻止事件的默认行为
</script>
